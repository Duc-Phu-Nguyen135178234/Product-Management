const listButtonStatus = document.querySelectorAll("[button-status]"); // create a list of button status dung queryselectorALL("[]") de chon attribute
if(listButtonStatus.length > 0) {     // choose queryselectorAll thi phai .length lam dieu kien >0 


  let url = new URL(window.location.href); // create a new URL copy from old url

  // Bắt sự kiện click gan lai thong so url
  listButtonStatus.forEach(button => {        //forEach all button => getAttribute assign into status
    button.addEventListener("click", () => {
      const status = button.getAttribute("button-status");

      if(status) {
        url.searchParams.set("status", status);    // if status have value set url.searchParams.set
      } else {
        url.searchParams.delete("status");          // delete
      }

      window.location.href = url.href;   //gan lai thong so tren url
    });
  });

  // Thêm class active mặc định    phan nay neu class active thi button se hien len
  const statusCurrent = url.searchParams.get("status") || "";

  const buttonCurrent = document.querySelector(`[button-status="${statusCurrent}"]`);
  
  if(buttonCurrent) {
    buttonCurrent.classList.add("active");
  }
}
